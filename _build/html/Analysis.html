lags = [7,14,21,28]

def peor_fold(lags, t):
    for lag in lags:
        ruta = f'data/{t}/figs/lag_{lag}/peor_rmse_lag_{lag}.png'
        img = mpimg.imread(ruta)
        plt.imshow(img)
        plt.axis('off')  # Oculta los ejes
        plt.show()
